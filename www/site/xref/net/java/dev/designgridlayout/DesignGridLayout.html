29  /**
30   * Swing LayoutManager that implements "Canonical Grids" as used by graphic 
31   * artists to design magazines, posters, forms...
32   * <p/>
33   * Canonical grids are useful for, but not limited to, all kinds of forms mostly 
34   * used in UI dialogs.
35   * <p/>
36   * DesignGridLayout provides a convenient API to have automatically well-balanced
37   * canonical grids panels. With this API, there is no need for any graphical
38   * designer.
39   * <p/>
40   * Typically, DesignGridLayout will be used as follows:
41   * <pre>
42   * public class MyPanel extends JPanel {
43   *     public MyPanel() {
44   *         DesignGridLayout layout = new DesignGridLayout(this);
45   *         setLayout(layout);
46   *         //...
47   *         layout.row().label(labelA).add(fieldA);
48   *         layout.row().label(labelB).add(fieldB);
49   *         //...
50   *         layout.centerRow().add(okButton, cancelButton);
51   *         }
52   *         //...
53   * </pre>
54   * 
55   * @author Jason Aaron Osgood
56   * @author Jean-Francois Poilpret
57   */
58  public class DesignGridLayout implements LayoutManager
59  {

235 	/* (non-Javadoc)
236 	 * @see java.awt.LayoutManager#layoutContainer
237 	 */
238 	public void layoutContainer(Container parent)
239 	{
240 		checkParent(parent);
241 
242 		// Make sure there's something to do
243 		if (_rowList.size() < 1)
244 		{
245 			return;
246 		}
247 
248 		// Always calculate the size of our contents
249 		layoutSize();
250 
251 		synchronized(parent.getTreeLock())
252 		{
253 			int x = left();
254 			int y = top();
255 			int rowWidth = parent.getSize().width - left() - right();
256 			for (AbstractRow row: _rowList)
257 			{
258 				if (row.items().size() > 0)
259 				{
260 					row.layoutRow(x, y, _hgap, rowWidth, _labelWidth);
261 				}
262 				y += row.height() + row.vgap();
263 			}
264 		}
265 	}

315 	private int getContainerGap(JComponent component, int position)
316 	{
317 		LayoutStyle layoutStyle = LayoutStyle.getSharedInstance();
318 		return layoutStyle.getContainerGap(component, position, _parent);
319 	}

321 	/*
322 	 * Horizontal gaps are easy.
323 	 * Since canonical grids are "balanced", just use the biggest intra-component 
324 	 * gap in the grid for all intra-component gaps.
325 	 * Vertical gaps are tougher.
326 	 * The vertical gaps between each component of the upper row and each 
327 	 * component of the lower row are compared. The heights of each component is 
328 	 * factored in, which seems to work well. 
329 	 * To be absolutely correct, each component's actual layout position should 
330 	 * be determined by factoring component heights, component baselines, and 
331 	 * each row's maximum height.
332 	 */
333 	private void computeGutters()
334 	{
335 		LayoutStyle layoutStyle = LayoutStyle.getSharedInstance();
336 
337 		// Handle horizontal gaps
338 		for (AbstractRow row: _rowList)
339 		{
340 			_hgap = Math.max(_hgap, row.hgap());
341 		}
342 
343 		// Vertical gaps (per row)
344 		for (int nthRow = 0; nthRow < (_rowList.size() - 1); nthRow++)
345 		{
346 			int maxComboHeight = 0;
347 			int rowGap = 0;
348 
349 			AbstractRow row = _rowList.get(nthRow);
350 			List<RowItem> items1 = row.items();
351 			List<RowItem> items2 = _rowList.get(nthRow + 1).items();
352 
353 			for (int nthItems1 = 0; nthItems1 < items1.size(); nthItems1++)
354 			{
355 				JComponent upper = items1.get(nthItems1).component();
356 				int aboveHeight = upper.getPreferredSize().height;
357 
358 				for (int nthItems2 = 0; nthItems2 < items2.size(); nthItems2++)
359 				{
360 					JComponent lower = items2.get(nthItems2).component();
361 					int belowHeight = lower.getPreferredSize().height;
362 
363 					int gap = layoutStyle.getPreferredGap(
364 						upper, lower, LayoutStyle.RELATED, SwingConstants.SOUTH, _parent);
365 					int comboHeight = aboveHeight + gap + belowHeight;
366 					if (comboHeight > maxComboHeight)
367 					{
368 						maxComboHeight = comboHeight;
369 						rowGap = gap;
370 					}
371 				}
372 			}
373 			row.vgap(rowGap);
374 		}
375 	}

377 	private Dimension layoutSize()
378 	{
379 		if (_layoutSize != null)
380 		{
381 			return _layoutSize;
382 		}
383 
384 		// Make sure there's something to do
385 		if (_rowList.size() < 1)
386 		{
387 			return new Dimension(0, 0);
388 		}
389 
390 		// Prior to computation start, reset everything and
391 		// calculate margins and gutters
392 		reset();
393 		computeMargins();
394 		computeGutters();
395 
396 		// First, reset each row and count the number of columns in the grid
397 		int maxColumns = countGridColumns();
398 
399 		// Second, calculate label width
400 		computeLabelWidth();
401 		
402 		// Third, compute width to use for columns of grid
403 		int maxWidth = maxGridRowsColumnWidth(maxColumns);
404 
405 		// Then, calculate the preferred total width
406 		int preferredWidth =
407 			maxWidth * maxColumns + left() + right() + (_hgap * (maxColumns - 1)) + 1;
408 
409 		// Account for labels
410 		if (_labelWidth > 0)
411 		{
412 			preferredWidth += _labelWidth;
413 			// Be sure to add hgap if needed
414 			if (maxColumns > 0)
415 			{
416 				preferredWidth += _hgap;
417 			}
418 		}
419 		
420 		// Don't forget to account for the minimum width of non grid rows
421 		preferredWidth = Math.max(preferredWidth, totalNonGridWidth());
422 
423 		// Total height
424 		int preferredHeight = totalHeight() + top() + bottom() + 1;
425 
426 		_layoutSize = new Dimension(preferredWidth, preferredHeight);
427 		return _layoutSize;
428 	}

430 	private int countGridColumns()
431 	{
432 		int maxColumns = 0;
433 		for (AbstractRow row: _rowList)
434 		{
435 			row.reset();
436 			// Note columns (sum item spans), not the count of components
437 			maxColumns = Math.max(maxColumns, row.gridColumns());
438 		}
439 		return maxColumns;
440 	}

462 	private int maxGridRowsColumnWidth(int maxColumns)
463 	{
464 		int maxWidth = 0;
465 		for (AbstractRow row: _rowList)
466 		{
467 			maxWidth = Math.max(maxWidth, row.maxColumnWidth(maxColumns));
468 		}

472 	private int totalNonGridWidth()
473 	{
474 		int maxWidth = 0;
475 		for (AbstractRow row: _rowList)
476 		{
477 			maxWidth = Math.max(maxWidth, row.totalWidth(_hgap));
478 		}
479 		return maxWidth + left() + right() + 1;
480 	}

482 	private void computeMargins()
483 	{
484 		// Make sure there's something to do
485 		if (_rowList.size() > 0)
486 		{
487 			// Handle top row
488 			computeTopMargin();
489 		
490 			// Handle bottom row
491 			computeBottomMargin();
492 		
493 			// Handle left-most and right-most columns
494 			computeLeftRightMargins();
495 		}
496 	}
497 	
498 	private void computeTopMargin()
499 	{
500 		AbstractRow topRow = _rowList.get(0);
501 		for (RowItem item: topRow.items())
502 		{
503 			int gap = getContainerGap(item.component(), SwingConstants.NORTH);
504 			_top = Math.max(_top, gap);
505 		}
506 	}
507 
508 	private void computeBottomMargin()
509 	{
510 		int maxComboHeight = 0;
511 		int bottomGap = 0;
512 		AbstractRow bottomRow = _rowList.get(_rowList.size() - 1);
513 		for (RowItem item: bottomRow.items())
514 		{
515 			JComponent c = item.component();
516 			int height = c.getPreferredSize().height;
517 			int gap = getContainerGap(c, SwingConstants.SOUTH);
518 			int comboHeight = height + gap;
519 			if (comboHeight > maxComboHeight)
520 			{
521 				maxComboHeight = comboHeight;
522 				bottomGap = gap;
523 			}
524 		}
525 		_bottom = Math.max(_bottom, bottomGap);
526 	}
527 	
528 	private void computeLeftRightMargins()
529 	{
530 		for (AbstractRow row: _rowList)
531 		{
532 			List<RowItem> items = row.items();
533 			if (items.size() > 0)
534 			{
535 				JComponent left = items.get(0).component();
536 				_left = Math.max(_left, getContainerGap(left, SwingConstants.WEST));
537 
538 				JComponent right = items.get(items.size() - 1).component();
539 				_right = Math.max(_right, getContainerGap(right, SwingConstants.EAST));
540 			}
541 		}
542 	}
