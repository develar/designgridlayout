30  /**
31   * Swing LayoutManager that implements "Canonical Grids" as used by graphic 
32   * artists to design magazines, posters, forms...
33   * <p/>
34   * Canonical grids are useful for, but not limited to, all kinds of forms mostly 
35   * used in UI dialogs.
36   * <p/>
37   * DesignGridLayout provides a convenient API to have automatically well-balanced
38   * canonical grids panels. With this API, there is no need for any graphical
39   * designer.
40   * <p/>
41   * Typically, DesignGridLayout will be used as follows:
42   * <pre>
43   * public class MyPanel extends JPanel {
44   *     public MyPanel() {
45   *         DesignGridLayout layout = new DesignGridLayout(this);
46   *         //...
47   *         layout.row().grid(labelA).add(fieldA);
48   *         layout.row().grid(labelB).add(fieldB);
49   *         //...
50   *         layout.row().center().add(okButton, cancelButton);
51   *         }
52   *         //...
53   * </pre>
54   * 
55   * @author Jason Aaron Osgood
56   * @author Jean-Francois Poilpret
57   */
58  public class DesignGridLayout implements LayoutManager
59  {
60  	/**
61  	 * Builds a DesignGridLayout instance attached to a {@link Container}.
62  	 * This instance should be then used to add rows and components to the parent
63  	 * container.
64  	 * <p/>
65  	 * Note that this constructor auomatically calls {@code parent.setLayout(this)}
66  	 * so you don't need to call it yourself.
67  	 * <p/>
68  	 * In no way should the {@link Container#add} and {@link Container#remove}
69  	 * ever be used with {@code parent}.
70  	 * 
71  	 * @param parent the container for which we want to use DesignGridLayout; 
72  	 * cannot be {@code null}.
73  	 */
74  	public DesignGridLayout(Container parent)
75  	{
76  		if (parent == null)
77  		{
78  			throw new NullPointerException("parent cannot be null");
79  		}
80  		_parent = parent;
81  		_parent.setLayout(this);
82  		_orientation = new OrientationPolicy(parent);
83  	}

85  	/**
86  	 * Define special margins ratios for the parent container. You normally
87  	 * won't use this method because DesignGridLayout uses the best margin values 
88  	 * for the current used Look and Feel. However, it may be useful when you use
89  	 * DesignGridLayout in special containers such as views used in a 
90  	 * docking-based application, in which case you would rather avoid wasting
91  	 * too much space as margins for each single docked view.
92  	 * 
93  	 * @param top top margin ratio that will be applied to the standard top
94  	 * margin for the current platform
95  	 * @param left left margin ratio that will be applied to the standard left
96  	 * margin for the current platform
97  	 * @param bottom bottom margin ratio that will be applied to the standard 
98  	 * bottom margin for the current platform
99  	 * @param right right margin ratio that will be applied to the standard 
100 	 * right margin for the current platform
101 	 * @return this instance of DesignGridLayout, allowing for chained calls
102 	 * to other methods (also known as "fluent API")
103 	 */
104 	public DesignGridLayout margins(double top, double left, double bottom, double right)
105 	{
106 		_topWeight = (top < 0.0 ? 0.0 : top);
107 		_leftWeight = (left < 0.0 ? 0.0 : left);
108 		_bottomWeight = (bottom < 0.0 ? 0.0 : bottom);
109 		_rightWeight = (right < 0.0 ? 0.0 : right);
110 		return this;
111 	}

152 	/**
153 	 * Creates a new row. The type of the row is not determined yet, but will
154 	 * be determined by the chained call performed on the returned 
155 	 * {@link IRowCreator}.
156 	 * <p/>
157 	 * The new row is located under the previously created row, which means that
158 	 * each line of code using this method creates a new row and all lines can 
159 	 * be read as defining the visual UI of the container.
160 	 * <p/>
161 	 * This method explicitly sets the vertical growth weight for this row; this 
162 	 * is applicable only if this row contains at least one component that can 
163 	 * vary in height (eg JScrollPane, vertical JSlider); if this row has no such
164 	 * component, then calling this method will have no impact, ie this row will
165 	 * always have a fixed height.
166 	 * <p/>
167 	 * Note that this method has no effect on the layout if there's no chained
168 	 * call on the returned {@link IRowCreator} (ie when you have code like
169 	 * {@code layout.row();}).
170 	 * <p/>
171 	 * <b>Important!</b> Note that it is generally useless to use this method:
172 	 * DesignGridLayout will by default assign a weight of 1.0 to all rows that 
173 	 * should have a variable height, hence during resize, extra height will be
174 	 * equally split to all such rows. Use this method only if you want a row to
175 	 * get more ({@code verticalWeight > 1.0}) or less 
176 	 * ({@code verticalWeight < 1.0}) extra height than other rows.
177 	 * 
178 	 * @param verticalWeight the weight given to this row when DesignGridLayout
179 	 * distributes extra height during resize actions; must be {@code >= 0.0} or
180 	 * the value will be ignored.
181 	 * @return a new {@code IRowCreator} that must be used to set the actual 
182 	 * type of the created row.
183 	 * @see #row()
184 	 */
185 	public IRowCreator row(double verticalWeight)
186 	{
187 		return new RowCreator(verticalWeight);
188 	}

191 	 * Adds a new empty row to the container. This row will never contain any 
192 	 * component and is used only for introducing vertical space between rows or 
193 	 * groups of rows. The height of that row is automatically calculated based
194 	 * on the fact that the previously-added row and the next-added row contain
195 	 * unrelated components.
196 	 */
197 	public void emptyRow()
198 	{
199 		if (!_rowList.isEmpty())
200 		{
201 			AbstractRow current = _rowList.get(_rowList.size() - 1);
202 			current.vgap(-1);
203 		}
204 	}

233 	/* (non-Javadoc)
234 	 * @see java.awt.LayoutManager#layoutContainer
235 	 */
236 	public void layoutContainer(Container parent)
237 	{
238 		checkParent(parent);
239 
240 		// Make sure there's something to do
241 		if (_rowList.isEmpty())
242 		{
243 			return;
244 		}
245 
246 		synchronized(parent.getTreeLock())
247 		{
248 			// Always calculate the size of our contents
249 			initialize();
250 
251 			// Calculate extra height to split between variable height rows
252 			double totalExtraHeight = 0.0;
253 			if (_totalWeight > 0.0)
254 			{
255 				totalExtraHeight = Math.max(
256 					0, (parent.getHeight() - _preferredSize.height) / _totalWeight);
257 			}
258 			
259 			// Check layout orientation
260 			boolean rtl = _orientation.isRightToLeft();
261 
262 			int x = left();
263 			int y = top();
264 			int parentWidth = parent.getWidth();
265 			// Never layout components smaller than the minimu size
266 			parentWidth = Math.max(parentWidth, _minimumSize.width);
267 
268 			int rowWidth = parentWidth - left() - right();
269 			// Calculate the total width assigned exclusively to sub-grids components
270 			int gridsWidth = rowWidth;
271 			// Exclude labels (and their gaps) from available space
272 			if (_totalLabelWidth > 0)
273 			{
274 				gridsWidth -= _totalLabelWidth + _maxGrids * _hgap;
275 			}
276 			// Exclude inter-grid gaps
277 			gridsWidth -= (_maxGrids - 1) * _gridgap;
278 			LayoutHelper helper = new LayoutHelper(_heightTester, parentWidth, rtl);
279 			for (AbstractRow row: _rowList)
280 			{
281 				helper.setY(y);
282 				int extraHeight = (int) (row.growWeight() * totalExtraHeight); 
283 				helper.setRowAvailableHeight(extraHeight + row.height());
284 				row.layoutRow(helper, x, _hgap, _gridgap, rowWidth, 
285 					gridsWidth, _labelWidths);
286 				y += row.height() + extraHeight + row.vgap();
287 			}
288 		}
289 	}

346 	private int getContainerGap(JComponent component, int position)
347 	{
348 		LayoutStyle layoutStyle = LayoutStyle.getSharedInstance();
349 		return layoutStyle.getContainerGap(component, position, _parent);
350 	}
351 	
352 	/*
353 	 * Horizontal gaps are easy.
354 	 * Since canonical grids are "balanced", just use the biggest intra-component 
355 	 * gap in the grid for all intra-component gaps.
356 	 * Vertical gaps are tougher.
357 	 * The vertical gaps between each component of the upper row and each 
358 	 * component of the lower row are compared. The heights of each component is 
359 	 * factored in, which seems to work well. 
360 	 */
361 	//FIXME vgaps calculation
362 	// To be absolutely correct, each component's actual layout position should
363 	// be determined by factoring component heights, component baselines, and 
364 	// each row's maximum height.
365 	// Problem is that this would have to be re-calculated every time... Maybe
366 	// it's better not to try to improve vgaps
367 	private void computeGutters()
368 	{
369 		LayoutStyle layoutStyle = LayoutStyle.getSharedInstance();
370 
371 		// Handle horizontal gaps
372 		_hgap = 0;
373 		_gridgap = 0;
374 		for (AbstractRow row: _rowList)
375 		{
376 			_hgap = Math.max(_hgap, row.hgap());
377 			_gridgap = Math.max(_gridgap, row.gridgap());
378 		}
379 
380 		// Vertical gaps (per row)
381 		for (int nthRow = 0; nthRow < (_rowList.size() - 1); nthRow++)
382 		{
383 			int maxComboHeight = 0;
384 			int rowGap = 0;
385 
386 			AbstractRow row = _rowList.get(nthRow);
387 			List<JComponent> items1 = row.components();
388 			List<JComponent> items2 = _rowList.get(nthRow + 1).components();
389 			int style = (row.vgap() == -1 ? LayoutStyle.UNRELATED : LayoutStyle.RELATED);
390 
391 			for (int nthItems1 = 0; nthItems1 < items1.size(); nthItems1++)
392 			{
393 				JComponent upper = items1.get(nthItems1);
394 				int aboveHeight = upper.getPreferredSize().height;
395 
396 				for (int nthItems2 = 0; nthItems2 < items2.size(); nthItems2++)
397 				{
398 					JComponent lower = items2.get(nthItems2);
399 					int belowHeight = lower.getPreferredSize().height;
400 
401 					int gap = layoutStyle.getPreferredGap(
402 						upper, lower, style, SwingConstants.SOUTH, _parent);
403 					int comboHeight = aboveHeight + gap + belowHeight;
404 					if (comboHeight > maxComboHeight)
405 					{
406 						maxComboHeight = comboHeight;
407 						rowGap = gap;
408 					}
409 				}
410 			}
411 			row.vgap(rowGap);
412 		}
413 		// Issue #24: force vgap of last row to 0 (might be -1)
414 		if (!_rowList.isEmpty())
415 		{
416 			_rowList.get(_rowList.size() - 1).vgap(0);
417 		}
418 	}

420 	private void initialize()
421 	{
422 		if (_preferredSize != null)
423 		{
424 			return;
425 		}
426 
427 		// Make sure there's something to do
428 		if (_rowList.isEmpty())
429 		{
430 			_preferredSize = new Dimension(0, 0);
431 			_minimumSize = new Dimension(0, 0);
432 			return;
433 		}
434 
435 		// Calculate margins and gutters
436 		computeMargins();
437 		computeGutters();
438 
439 		// Now, initialize each row and count the number of grids & columns 
440 		// in the grids
441 		for (AbstractRow row: _rowList)
442 		{
443 			row.init();
444 		}
445 		countGrids();
446 
447 		// Second, calculate labels width
448 		computeLabelWidths();
449 
450 		// Compute preferred & minimum widths for each sub-grid (without labels), 
451 		// use largest width for all grids
452 		int preferredWidth = computeGridWidth(PrefWidthExtractor.INSTANCE);
453 		int minimumWidth = computeGridWidth(MinWidthExtractor.INSTANCE);
454 
455 		// Total height
456 		int preferredHeight = totalHeight() + top() + bottom() + 1;
457 
458 		// Calculate total height growth factor of all variable height rows
459 		_totalWeight = 0.0;
460 		for (AbstractRow row: _rowList)
461 		{
462 			_totalWeight += row.growWeight();
463 		}
464 
465 		_preferredSize = new Dimension(preferredWidth, preferredHeight);
466 		_minimumSize = new Dimension(minimumWidth, preferredHeight);
467 	}
468 	
469 	private void countGrids()
470 	{
471 		// Calculate the actual number of sub-grids
472 		_maxGrids = 0;
473 		for (AbstractRow row: _rowList)
474 		{
475 			_maxGrids = Math.max(_maxGrids, row.numGrids());
476 		}
477 		// Inform each row about the total number of sub-grids
478 		for (AbstractRow row: _rowList)
479 		{
480 			row.totalGrids(_maxGrids);
481 		}
482 	}
483 
484 	private int countGridColumns(int grid)
485 	{
486 		int maxColumns = 0;
487 		for (AbstractRow row: _rowList)
488 		{
489 			// Note columns (sum item spans), not the count of components
490 			maxColumns = Math.max(maxColumns, row.gridColumns(grid));
491 		}
492 		return maxColumns;
493 	}
494 	
495 	private void computeLabelWidths()
496 	{
497 		_labelWidths.clear();
498 		_totalLabelWidth = 0;
499 		for (int i = 0; i < _maxGrids; i++)
500 		{
501 			int width = 0;
502 			for (AbstractRow row: _rowList)
503 			{
504 				// Label width first
505 				width = Math.max(width, row.labelWidth(i));
506 			}
507 			_labelWidths.add(width);
508 			_totalLabelWidth += width;
509 		}
510 	}
511 
512 	private int totalHeight()
513 	{
514 		int totalHeight = 0;
515 		for (AbstractRow row: _rowList)
516 		{
517 			totalHeight += row.height() + row.vgap();
518 		}
519 		return totalHeight;
520 	}

522 	private int computeGridWidth(IExtractor extractor)
523 	{
524 		// Compute preferred width for each sub-grid (without labels), 
525 		// use largest width for all grids
526 		int width = 0;
527 		for (int grid = 0; grid < _maxGrids; grid++)
528 		{
529 			int maxColumns = countGridColumns(grid);
530 	
531 			// Compute width to use for columns of grid
532 			int maxWidth = maxGridRowsColumnWidth(grid, maxColumns, extractor);
533 	
534 			// Then, calculate the preferred width for that grid
535 			int gridWidth = maxWidth * maxColumns + (_hgap * (maxColumns - 1)) + 1;
536 	
537 			width = Math.max(width, gridWidth);
538 		}
539 		// Now use preferred width for each subgrid
540 		width *= _maxGrids;
541 		
542 		// Account for labels
543 		if (_totalLabelWidth > 0)
544 		{
545 			width += _totalLabelWidth + (_maxGrids * _hgap);
546 		}
547 
548 		// Add gaps between grids
549 		// first hgap is correct (label to 1st comp), 2nd gap should be bigger
550 		width += (_maxGrids - 1) * _gridgap;
551 		
552 		// Add left and right margins
553 		width += left() + right();
554 
555 		// Don't forget to account for the minimum width of non grid rows
556 		width = Math.max(width, totalNonGridWidth(extractor));
557 		
558 		return width;
559 	}
560 	
561 	private int maxGridRowsColumnWidth(int grid, int maxColumns, IExtractor extractor)
562 	{
563 		int maxWidth = 0;
564 		for (AbstractRow row: _rowList)
565 		{
566 			int width = row.maxColumnWidth(grid, maxColumns, extractor);
567 			// If current grid spans several sub-grids, then colum width must
568 			// be reduced (by removing extra labels) and divide result by number
569 			// of spanned sub-grids
570 			int span = row.gridspan(grid);
571 			if (span > 1)
572 			{
573 				for (int i = 1; i < span; i++)
574 				{
575 					width -= _gridgap + _labelWidths.get(grid + i) + _hgap;
576 				}
577 				int fudge = width % span;
578 				width /= span;
579 				if (fudge > 0)
580 				{
581 					width += 1;
582 				}
583 			}
584 			maxWidth = Math.max(maxWidth, width);
585 		}
586 		return maxWidth;
587 	}
588 
589 	private int totalNonGridWidth(IExtractor extractor)
590 	{
591 		int maxWidth = 0;
592 		for (AbstractRow row: _rowList)
593 		{
594 			maxWidth = Math.max(maxWidth, row.totalNonGridWidth(_hgap, extractor));
595 		}
596 		return maxWidth + left() + right() + 1;
597 	}

599 	private void computeMargins()
600 	{
601 		// Handle top row
602 		computeTopMargin();
603 	
604 		// Handle bottom row
605 		computeBottomMargin();
606 	
607 		// Handle left-most and right-most columns
608 		computeLeftRightMargins();
609 	}
610 	
611 	private void computeTopMargin()
612 	{
613 		_top = 0;
614 		AbstractRow topRow = _rowList.get(0);
615 		for (JComponent component: topRow.components())
616 		{
617 			int gap = getContainerGap(component, SwingConstants.NORTH);
618 			_top = Math.max(_top, gap);
619 		}
620 	}
621 
622 	private void computeBottomMargin()
623 	{
624 		_bottom = 0;
625 		int maxComboHeight = 0;
626 		int bottomGap = 0;
627 		AbstractRow bottomRow = _rowList.get(_rowList.size() - 1);
628 		for (JComponent component: bottomRow.components())
629 		{
630 			int height = component.getPreferredSize().height;
631 			int gap = getContainerGap(component, SwingConstants.SOUTH);
632 			int comboHeight = height + gap;
633 			if (comboHeight > maxComboHeight)
634 			{
635 				maxComboHeight = comboHeight;
636 				bottomGap = gap;
637 			}
638 		}
639 		_bottom = Math.max(_bottom, bottomGap);
640 	}
641 	
642 	private void computeLeftRightMargins()
643 	{
644 		_left = 0;
645 		_right = 0;
646 		for (AbstractRow row: _rowList)
647 		{
648 			List<JComponent> components = row.components();
649 			if (!components.isEmpty())
650 			{
651 				JComponent left = components.get(0);
652 				_left = Math.max(_left, getContainerGap(left, SwingConstants.WEST));
653 
654 				JComponent right = components.get(components.size() - 1);
655 				_right = Math.max(_right, getContainerGap(right, SwingConstants.EAST));
656 			}
657 		}
658 	}
