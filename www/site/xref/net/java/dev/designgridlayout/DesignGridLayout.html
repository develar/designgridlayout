31  /**
32   * Swing LayoutManager that implements "Canonical Grids" as used by graphic 
33   * artists to design magazines, posters, forms...
34   * <p/>
35   * Canonical grids are useful for, but not limited to, all kinds of forms mostly 
36   * used in UI dialogs.
37   * <p/>
38   * DesignGridLayout provides a convenient API to have automatically well-balanced
39   * canonical grids panels. With this API, there is no need for any graphical
40   * designer.
41   * <p/>
42   * Typically, DesignGridLayout will be used as follows:
43   * <pre>
44   * public class MyPanel extends JPanel {
45   *     public MyPanel() {
46   *         DesignGridLayout layout = new DesignGridLayout(this);
47   *         //...
48   *         layout.row().grid(labelA).add(fieldA);
49   *         layout.row().grid(labelB).add(fieldB);
50   *         //...
51   *         layout.row().center().add(okButton, cancelButton);
52   *         }
53   *         //...
54   * </pre>
55   * 
56   * @author Jason Aaron Osgood
57   * @author Jean-Francois Poilpret
58   */
59  public class DesignGridLayout implements LayoutManager
60  {
61  	/**
62  	 * Builds a DesignGridLayout instance attached to a {@link Container}.
63  	 * This instance should be then used to add rows and components to the parent
64  	 * container.
65  	 * <p/>
66  	 * Note that this constructor auomatically calls {@code parent.setLayout(this)}
67  	 * so you don't need to call it yourself.
68  	 * <p/>
69  	 * In no way should the {@link Container#add} and {@link Container#remove}
70  	 * ever be used with {@code parent}.
71  	 * 
72  	 * @param parent the container for which we want to use DesignGridLayout; 
73  	 * cannot be {@code null}.
74  	 */
75  	public DesignGridLayout(Container parent)
76  	{
77  		if (parent == null)
78  		{
79  			throw new NullPointerException("parent cannot be null");
80  		}
81  		_parent = parent;
82  		_parent.setLayout(this);
83  		_orientation = new OrientationPolicy(parent);
84  	}

86  	/**
87  	 * Define special margins ratios for the parent container. You normally
88  	 * won't use this method because DesignGridLayout uses the best margin values 
89  	 * for the current used Look and Feel. However, it may be useful when you use
90  	 * DesignGridLayout in special containers such as views used in a 
91  	 * docking-based application, in which case you would rather avoid wasting
92  	 * too much space as margins for each single docked view.
93  	 * 
94  	 * @param top top margin ratio that will be applied to the standard top
95  	 * margin for the current platform
96  	 * @param left left margin ratio that will be applied to the standard left
97  	 * margin for the current platform
98  	 * @param bottom bottom margin ratio that will be applied to the standard 
99  	 * bottom margin for the current platform
100 	 * @param right right margin ratio that will be applied to the standard 
101 	 * right margin for the current platform
102 	 * @return {@code this} instance of DesignGridLayout, allowing for chained 
103 	 * calls to other methods (also known as "fluent API")
104 	 */
105 	public DesignGridLayout margins(double top, double left, double bottom, double right)
106 	{
107 		_topWeight = (top < 0.0 ? 0.0 : top);
108 		_leftWeight = (left < 0.0 ? 0.0 : left);
109 		_bottomWeight = (bottom < 0.0 ? 0.0 : bottom);
110 		_rightWeight = (right < 0.0 ? 0.0 : right);
111 		return this;
112 	}

153 	/**
154 	 * Creates a new row. The type of the row is not determined yet, but will
155 	 * be determined by the chained call performed on the returned 
156 	 * {@link IRowCreator}.
157 	 * <p/>
158 	 * The new row is located under the previously created row, which means that
159 	 * each line of code using this method creates a new row and all lines can 
160 	 * be read as defining the visual UI of the container.
161 	 * <p/>
162 	 * This method explicitly sets the vertical growth weight for this row; this 
163 	 * is applicable only if this row contains at least one component that can 
164 	 * vary in height (eg JScrollPane, vertical JSlider); if this row has no such
165 	 * component, then calling this method will have no impact, ie this row will
166 	 * always have a fixed height.
167 	 * <p/>
168 	 * Note that this method has no effect on the layout if there's no chained
169 	 * call on the returned {@link IRowCreator} (ie when you have code like
170 	 * {@code layout.row();}).
171 	 * <p/>
172 	 * <b>Important!</b> Note that it is generally useless to use this method:
173 	 * DesignGridLayout will by default assign a weight of 1.0 to all rows that 
174 	 * should have a variable height, hence during resize, extra height will be
175 	 * equally split to all such rows. Use this method only if you want a row to
176 	 * get more ({@code verticalWeight > 1.0}) or less 
177 	 * ({@code verticalWeight < 1.0}) extra height than other rows.
178 	 * 
179 	 * @param verticalWeight the weight given to this row when DesignGridLayout
180 	 * distributes extra height during resize actions; must be {@code >= 0.0} or
181 	 * the value will be ignored.
182 	 * @return a new {@code IRowCreator} that must be used to set the actual 
183 	 * type of the created row.
184 	 * @see #row()
185 	 */
186 	public IRowCreator row(double verticalWeight)
187 	{
188 		return new RowCreator(verticalWeight);
189 	}

234 	/* (non-Javadoc)
235 	 * @see java.awt.LayoutManager#layoutContainer
236 	 */
237 	public void layoutContainer(Container parent)
238 	{
239 		checkParent(parent);
240 
241 		// Make sure there's something to do
242 		if (_rows.isEmpty())
243 		{
244 			return;
245 		}
246 
247 		synchronized(parent.getTreeLock())
248 		{
249 			// Always calculate the size of our contents
250 			initialize();
251 
252 			// Calculate extra height to split between variable height rows
253 			double totalExtraHeight = 0.0;
254 			if (_totalWeight > 0.0)
255 			{
256 				totalExtraHeight = Math.max(
257 					0, (parent.getHeight() - _preferredSize.height) / _totalWeight);
258 			}
259 			
260 			// Check layout orientation
261 			boolean rtl = _orientation.isRightToLeft();
262 
263 			int x = left();
264 			int y = top();
265 			int parentWidth = parent.getWidth();
266 			// Never layout components smaller than the minimu size
267 			parentWidth = Math.max(parentWidth, _minimumSize.width);
268 
269 			int rowWidth = parentWidth - left() - right();
270 			// Calculate the total width assigned exclusively to sub-grids components
271 			int gridsWidth = rowWidth;
272 			// Exclude labels (and their gaps) from available space
273 			if (_totalLabelWidth > 0)
274 			{
275 				gridsWidth -= _totalLabelWidth + _maxGrids * _hgap;
276 			}
277 			// Exclude inter-grid gaps
278 			gridsWidth -= (_maxGrids - 1) * _gridgap;
279 			
280 			// Start laying out every single row (all components but row-span ones)
281 			LayoutHelper helper = new LayoutHelper(_heightTester, parentWidth, rtl, _rows);
282 			for (AbstractRow row: _rows)
283 			{
284 				// Issue #30 - check that row is not empty
285 				if (!row.isEmpty())
286 				{
287 					helper.setY(y);
288 					int extraHeight = (int) (row.growWeight() * totalExtraHeight); 
289 					helper.setRowAvailableHeight(extraHeight + row.height());
290 					row.layoutRow(helper, x, _hgap, _gridgap, rowWidth, 
291 						gridsWidth, _labelWidths);
292 					row.actualHeight(row.height() + extraHeight);
293 					y += row.actualHeight() + row.vgap();
294 				}
295 			}
296 			
297 			// Second pass: all row-span components
298 			int rowIndex = 0;
299 			for (AbstractRow row: _rows)
300 			{
301 				for (IRowItem item: row.items())
302 				{
303 					if (item.isFirstSpanRow() && !item.isLastSpanRow())
304 					{
305 						// Calculate size based on number of spanned rows
306 						helper.setHeight(rowIndex, item.component(), item.rowSpan());
307 					}
308 				}
309 				rowIndex++;
310 			}
311 
312 		}
313 	}

368 	private int getContainerGap(JComponent component, int position)
369 	{
370 		LayoutStyle layoutStyle = LayoutStyle.getSharedInstance();
371 		return layoutStyle.getContainerGap(component, position, _parent);
372 	}

374 	/*
375 	 * Horizontal gaps are easy.
376 	 * Since canonical grids are "balanced", just use the biggest intra-component 
377 	 * gap in the grid for all intra-component gaps.
378 	 * Vertical gaps are tougher.
379 	 * The vertical gaps between each component of the upper row and each 
380 	 * component of the lower row are compared. The heights of each component is 
381 	 * factored in, which seems to work well. 
382 	 */
383 	//FIXME vgaps calculation
384 	// To be absolutely correct, each component's actual layout position should
385 	// be determined by factoring component heights, component baselines, and 
386 	// each row's maximum height.
387 	// Problem is that this would have to be re-calculated every time... Maybe
388 	// it's better not to try to improve vgaps
389 	private void computeGutters()
390 	{
391 		LayoutStyle layoutStyle = LayoutStyle.getSharedInstance();
392 
393 		// Handle horizontal gaps
394 		_hgap = 0;
395 		_gridgap = 0;
396 		for (AbstractRow row: _rows)
397 		{
398 			_hgap = Math.max(_hgap, row.hgap());
399 			_gridgap = Math.max(_gridgap, row.gridgap());
400 		}
401 
402 		// Vertical gaps (per row)
403 		int nthRow = 0;
404 		for (AbstractRow row: _rows)
405 		{
406 			nthRow++;
407 			if (row.isEmpty())
408 			{
409 				// Current row has no component: nothing to compute!
410 				continue;
411 			}
412 			AbstractRow next = nextNonEmptyRow(nthRow);
413 			if (next == null)
414 			{
415 				// Current row is the last row with components: 
416 				// computation is finished
417 				break;
418 			}
419 
420 			int maxComboHeight = 0;
421 			int rowGap = 0;
422 
423 			List<? extends IRowItem> items1 = row.allItems();
424 			List<? extends IRowItem> items2 = next.allItems();
425 			int style = (row.hasUnrelatedGap() ? LayoutStyle.UNRELATED : LayoutStyle.RELATED);
426 
427 			for (IRowItem item1: items1)
428 			{
429 				if (item1.isLastSpanRow())
430 				{
431 					JComponent upper = item1.component();
432 					int aboveHeight = upper.getPreferredSize().height;
433 
434 					for (IRowItem item2: items2)
435 					{
436 						if (item2.isFirstSpanRow())
437 						{
438 							JComponent lower = item2.component();
439 							int belowHeight = lower.getPreferredSize().height;
440 		
441 							int gap = layoutStyle.getPreferredGap(
442 								upper, lower, style, SwingConstants.SOUTH, _parent);
443 							int comboHeight = aboveHeight + gap + belowHeight;
444 							if (comboHeight > maxComboHeight)
445 							{
446 								maxComboHeight = comboHeight;
447 								rowGap = gap;
448 							}
449 						}
450 					}
451 				}
452 			}
453 			row.vgap(rowGap);
454 		}
455 	}

457 	private void initialize()
458 	{
459 		if (_preferredSize != null)
460 		{
461 			return;
462 		}
463 
464 		// Make sure there's something to do
465 		if (_rows.isEmpty())
466 		{
467 			_preferredSize = new Dimension(0, 0);
468 			_minimumSize = new Dimension(0, 0);
469 			return;
470 		}
471 		
472 		// First of all count number of canonical grids in the whole panel
473 		countGrids();
474 
475 		// Check all spanRow() calls are correct on all GridRows
476 		// (show colored placeholder in all locations where a problem occurs)
477 		for (AbstractRow row: _rows)
478 		{
479 			row.checkSpanRows();
480 		}
481 		
482 		// Calculate margins and gutters
483 		computeMargins();
484 		computeGutters();
485 		
486 		// Initialize the list of all rowspan items across the layout
487 		initRowSpanItems();
488 
489 		// Initialize each row (compute width, height, baseline)
490 		for (AbstractRow row: _rows)
491 		{
492 			row.init();
493 		}
494 
495 		// Calculate labels width for all grids
496 		computeLabelWidths();
497 
498 		// Compute preferred & minimum widths for each sub-grid (without labels), 
499 		// use largest width for all grids
500 		int preferredWidth = computeGridWidth(PrefWidthExtractor.INSTANCE);
501 		int minimumWidth = computeGridWidth(MinWidthExtractor.INSTANCE);
502 
503 		// Total height
504 		int preferredHeight = totalHeight() + top() + bottom() + 1;
505 
506 		// Calculate total height growth factor of all variable height rows
507 		_totalWeight = 0.0;
508 		for (AbstractRow row: _rows)
509 		{
510 			_totalWeight += row.growWeight();
511 		}
512 
513 		_preferredSize = new Dimension(preferredWidth, preferredHeight);
514 		_minimumSize = new Dimension(minimumWidth, preferredHeight);
515 	}
516 
517 	private void initRowSpanItems()
518 	{
519 		// Perform pre-calculation of preferred height per row for each spanned item
520 		int rowIndex = 0;
521 		for (AbstractRow row: _rows)
522 		{
523 			for (IRowItem item: row.items())
524 			{
525 				if (item.isFirstSpanRow())
526 				{
527 					List<AbstractRow> spannedRows = 
528 						_rows.subList(rowIndex, rowIndex + item.rowSpan());
529 					item.setSpannedRows(spannedRows);
530 				}
531 			}
532 			rowIndex++;
533 		}
534 	}

536 	private void countGrids()
537 	{
538 		// Calculate the actual number of sub-grids
539 		_maxGrids = 0;
540 		for (AbstractRow row: _rows)
541 		{
542 			_maxGrids = Math.max(_maxGrids, row.numGrids());
543 		}
544 		// Inform each row about the total number of sub-grids
545 		for (AbstractRow row: _rows)
546 		{
547 			row.totalGrids(_maxGrids);
548 		}
549 	}

551 	private int countGridColumns(int grid)
552 	{
553 		int maxColumns = 0;
554 		for (AbstractRow row: _rows)
555 		{
556 			// Note columns (sum item spans), not the count of components
557 			maxColumns = Math.max(maxColumns, row.gridColumns(grid));
558 		}
559 		return maxColumns;
560 	}

562 	private void computeLabelWidths()
563 	{
564 		_labelWidths.clear();
565 		_totalLabelWidth = 0;
566 		for (int i = 0; i < _maxGrids; i++)
567 		{
568 			int width = 0;
569 			for (AbstractRow row: _rows)
570 			{
571 				// Label width first
572 				width = Math.max(width, row.labelWidth(i));
573 			}
574 			_labelWidths.add(width);
575 			_totalLabelWidth += width;
576 		}
577 	}

579 	private int totalHeight()
580 	{
581 		int totalHeight = 0;
582 		for (AbstractRow row: _rows)
583 		{
584 			totalHeight += row.height() + row.vgap();
585 		}
586 		return totalHeight;
587 	}

589 	private int computeGridWidth(IExtractor extractor)
590 	{
591 		// Compute preferred width for each sub-grid (without labels), 
592 		// use largest width for all grids
593 		int width = 0;
594 		for (int grid = 0; grid < _maxGrids; grid++)
595 		{
596 			int maxColumns = countGridColumns(grid);
597 	
598 			// Compute width to use for columns of grid
599 			int maxWidth = maxGridRowsColumnWidth(grid, maxColumns, extractor);
600 	
601 			// Then, calculate the preferred width for that grid
602 			int gridWidth = maxWidth * maxColumns + (_hgap * (maxColumns - 1)) + 1;
603 	
604 			width = Math.max(width, gridWidth);
605 		}
606 		// Now use preferred width for each subgrid
607 		width *= _maxGrids;
608 		
609 		// Account for labels
610 		if (_totalLabelWidth > 0)
611 		{
612 			width += _totalLabelWidth + (_maxGrids * _hgap);
613 		}
614 
615 		// Add gaps between grids
616 		// first hgap is correct (label to 1st comp), 2nd gap should be bigger
617 		width += (_maxGrids - 1) * _gridgap;
618 		
619 		// Add left and right margins
620 		width += left() + right();
621 
622 		// Don't forget to account for the minimum width of non grid rows
623 		width = Math.max(width, totalNonGridWidth(extractor));
624 		
625 		return width;
626 	}
627 	
628 	private int maxGridRowsColumnWidth(int grid, int maxColumns, IExtractor extractor)
629 	{
630 		int maxWidth = 0;
631 		for (AbstractRow row: _rows)
632 		{
633 			int width = row.maxColumnWidth(grid, maxColumns, extractor);
634 			// If current grid spans several sub-grids, then colum width must
635 			// be reduced (by removing extra labels) and divide result by number
636 			// of spanned sub-grids
637 			int span = row.gridspan(grid);
638 			if (span > 1)
639 			{
640 				for (int i = 1; i < span; i++)
641 				{
642 					width -= _gridgap + _labelWidths.get(grid + i) + _hgap;
643 				}
644 				int fudge = width % span;
645 				width /= span;
646 				if (fudge > 0)
647 				{
648 					width += 1;
649 				}
650 			}
651 			maxWidth = Math.max(maxWidth, width);
652 		}
653 		return maxWidth;
654 	}
655 
656 	private int totalNonGridWidth(IExtractor extractor)
657 	{
658 		int maxWidth = 0;
659 		for (AbstractRow row: _rows)
660 		{
661 			maxWidth = Math.max(maxWidth, row.totalNonGridWidth(_hgap, extractor));
662 		}
663 		return maxWidth + left() + right() + 1;
664 	}

667 	{
668 		// Handle top row
669 		computeTopMargin();
670

672 		computeBottomMargin();
673 	
674 		// Handle left-most and right-most columns
675 		computeLeftRightMargins();
676 	}

678 	private void computeTopMargin()
679 	{
680 		_top = 0;
681 		// Issue #30 - find the first non empty row
682 		AbstractRow topRow = firstNonEmptyRow();
683 		if (topRow != null)
684 		{
685 			for (IRowItem item: topRow.allItems())
686 			{
687 				int gap = getContainerGap(item.component(), SwingConstants.NORTH);
688 				_top = Math.max(_top, gap);
689 			}
690 		}
691 	}
692 
693 	private void computeBottomMargin()
694 	{
