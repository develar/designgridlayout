140 	private ISpannableGridRow newGrid(JLabel label, int gridspan)
141 	{
142 		// Fix the span of the previous sub-grid (if it was in auto-span mode)
143 		if (_current != null)
144 		{
145 			_current.gridspan(1);
146 			_gridIndex += _current.gridspan();
147 		}
148 		// Find the matching subgrid (if any) of the previous row
149 		SubGrid previous = (_previous != null ? _previous.findGrid(_gridIndex) : null);
150 		// Initialize the sublist of _items to be used by the new subgrid
151 		_current = new SubGrid(new SubList(_items), previous, parent(), label, gridspan);
152 		_grids.add(_current);
153 		return this;
154 	}

184 	@Override void totalGrids(int totalGrids)
185 	{
186 		// Set the correct grid-span for the last sub-grid if needed
187 		int span = totalGrids;
188 		Iterator<SubGrid> i = _grids.iterator();
189 		while (i.hasNext())
190 		{
191 			SubGrid grid = i.next();
192 			if (i.hasNext())
193 			{
194 				span -= grid.gridspan();
195 			}
196 			else
197 			{
198 				grid.gridspan(span);
199 			}
200 		}
201 		// Add as many fake grids as necessary to the sub-grids list
202 		_totalGrids = totalGrids;
203 	}

225 	@Override int gridgap()
226 	{
227 		ComponentGapsHelper helper = ComponentGapsHelper.instance();
228 		int gridgap = 0;
229 		for (int i = 0; i < _grids.size() - 1; i++)
230 		{
231 			List<RowItem> leftGrid = _grids.get(i).items();
232 			List<RowItem> rightGrid = _grids.get(i + 1).items();
233 			if (!(leftGrid.isEmpty() || rightGrid.isEmpty()))
234 			{
235 				JComponent left = leftGrid.get(leftGrid.size() - 1).component();
236 				JComponent right = rightGrid.get(0).component();
237 				int gap = helper.getHorizontalGap(
238 					left, right, LayoutStyle.UNRELATED, parent());
239 				gridgap = Math.max(gridgap, gap);
240 			}
241 		}
242 		return gridgap;
243 	}

285 	@Override int layoutRow(LayoutHelper helper, int left, int hgap, int gridgap, 
286 		int rowWidth, int gridsWidth, List<Integer> labelsWidth)
287 	{
288 		int x = left;
289 		int actualHeight = 0;
290 		int gridWidth = gridsWidth / _totalGrids;
291 		int gridIndex = 0;
292 		Iterator<Integer> label = labelsWidth.iterator();
293 		for (SubGrid grid: _grids)
294 		{
295 			// Find the label for the current sub-grid
296 			int labelWidth = label.next();
297 
298 			// Calculate the actual width for this sub-grid (depends on spans)
299 			int width = gridWidth;
300 			for (int i = 1; i < grid.gridspan(); i++)
301 			{
302 				width += gridgap + label.next() + hgap + gridWidth;
303 			}
304 			gridIndex += grid.gridspan();
305 			if (gridIndex == _totalGrids)
306 			{
307 				// Add fudge to the true last grid 
308 				// Note that this is not necessarily the last grid of _grids!
309 				int fudge = gridsWidth % _totalGrids;
310 				width += fudge;
311 			}
312 			
313 			// Layout the sub-grid
314 			int height = grid.layoutRow(
315 				helper, x, height(), baseline(), hgap, width, labelWidth);
316 			actualHeight = Math.max(actualHeight, height);
317 
318 			// Position origin to the next grid
319 			x += labelWidth + hgap + width + gridgap;
320 		}
321 		return actualHeight;
322 	}

324 	SubGrid findGrid(int index)
325 	{
326 		int i = 0;
327 		for (SubGrid grid: _grids)
328 		{
329 			if (i == index)
330 			{
331 				return grid;
332 			}
333 			i += grid.gridspan();
334 			if (i > index)
335 			{
336 				break;
337 			}
338 		}
339 		return null;
340 	}
