41  	public void go(boolean exitOnClose)
42  	{
43  		_frame = new JFrame(name());
44  		_frame.setName(getClass().getSimpleName());
45  
46  		_frame.setDefaultCloseOperation(exitOnClose
47  			? JFrame.EXIT_ON_CLOSE
48  			: WindowConstants.DISPOSE_ON_CLOSE);
49  		JPanel top = new JPanel();
50  		DesignGridLayout layout = new DesignGridLayout(top);
51  		top.setName("TOP");
52  
53  		build(layout);
54  
55  		addTopPanel(top);
56  		prePack();
57  		_frame.pack();
58  		_frame.setLocationRelativeTo(null);
59  		_frame.setVisible(true);
60  	}
