17  import javax.swing.JSeparator;
18  
19  // Demonstrate bad calls to spanRow()
20  public class RowSpan4ErrorMarkers extends AbstractBaseExample
21  {
22  	public static void main(String[] args)
23  	{
24  		RowSpan4ErrorMarkers example = new RowSpan4ErrorMarkers();
25  		example.go(true);
26  	}
27  
28  	@Override public void build(DesignGridLayout layout)
29  	{
30  		// spanrow() called in first row
31  		layout.row().grid(label("lbl0")).spanRow();
32  		layout.row().center().fill().add(new JSeparator());
33  
34  		// spanrow() called on a subgrid with different number of columns
35  		layout.row().grid(label("lbl1")).add(field("field1"), field("field1"), field("field1"));
36  		layout.row().grid(label("lbl2")).add(field("field2")).spanRow();
37  		layout.row().center().fill().add(new JSeparator());
38  		
39  		// spanrow() called with no matching item
40  		layout.row().grid(label("lbl1")).add(field("field1"), 2);
41  		layout.row().grid(label("lbl2")).add(field("field2")).spanRow();
42  		layout.row().center().fill().add(new JSeparator());
43  		
44  		// spanrow() called with no matching item
45  		layout.row().grid(label("lbl1")).add(field("field1"), 2);
46  		layout.row().grid(label("lbl2")).spanRow().add(field("field2"));
47  		layout.row().center().fill().add(new JSeparator());
48  
49  		// spanrow() called with no matching item
50  		layout.row().grid(label("lbl1")).add(field("field1"), 2).add(field("field1"));
51  		layout.row().grid(label("lbl2")).spanRow();
52  		layout.row().center().fill().add(new JSeparator());
53  
54  		// spanrow() called on non matching subgrids
55  		layout.row().grid(label("lbl11"), 2).add(field("field1")).grid(label("lbl13")).add(field("field1"));
56  		layout.row().grid(label("lbl21"), 1).add(field("field2")).grid(label("lbl22")).spanRow();
57  		layout.row().center().fill().add(new JSeparator());
58  		
59  		// spanrow() called while the previous row is not a gridrow
60  		layout.row().left().add(button());
61  		layout.row().grid(label("lbl1")).spanRow();
62  		layout.row().center().fill().add(new JSeparator());
63  		
64  		// multiple spanrow() in consecutive gridrows, the first being bad
65  		// Note: the red marker must span rows 2 & 3
66  		layout.row().grid(label("lbl1")).add(field("field1"), field("field1"), field("field1"));
67  		layout.row().grid(label("lbl2")).add(field("field2")).spanRow();
68  		layout.row().grid(label("lbl3")).add(field("field3")).spanRow();
69  	}
