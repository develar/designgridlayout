81  	private class ShowHideAction implements ActionListener
82  	{
83  		public ShowHideAction(IRow row, boolean show)
84  		{
85  			_row = row;

87  		}
88  		
89  		@Override public void actionPerformed(ActionEvent evt)
90  		{
91  			if (_show)
92  			{
93  				_row.show();

95  			else
96  			{
97  				_row.hide();

99  			frame().pack();
100 		}
101
