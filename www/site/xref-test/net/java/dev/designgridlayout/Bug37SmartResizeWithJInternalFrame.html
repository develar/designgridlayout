44  	public void go(boolean exitOnClose)
45  	{
46  		_frame = new JFrame();
47  		_frame.setName(getClass().getSimpleName());
48  		_frame.setBounds(30, 30, 800, 600);
49  		_frame.setDefaultCloseOperation(exitOnClose
50  			? JFrame.EXIT_ON_CLOSE
51  			: WindowConstants.DISPOSE_ON_CLOSE);
52  		JDesktopPane desktop = new JDesktopPane();
53  		_frame.setContentPane(desktop);
54  		_frame.setVisible(true);
55  		
56  		JInternalFrame frame = new JInternalFrame("", true, true, true, true);
57  		frame.setName("INTERNAL");
58  		frame.setLocation(30, 30);
59  		
60  		JPanel top = new JPanel();
61  		DesignGridLayout layout = new DesignGridLayout(top);
62  		top.setName("TOP");
63  		build(layout);
64  		frame.setContentPane(top);
65  
66  		// Order of calling desktop.add() is important wrt frame.pack() call!
67  		desktop.add(frame);
68  		frame.pack();
69  		frame.setVisible(true);
70  		try
71  		{
72  			frame.setSelected(true);
73  		}
74  		catch (java.beans.PropertyVetoException e)
75  		{
76  		}
77  	}

79  	protected void build(DesignGridLayout layout)

92  	static protected void setTableHeight(JTable table, int rows)
93  	{
94  		int width = table.getColumnModel().getTotalColumnWidth();
95  //		int width = 600;
96  		int height = rows * table.getRowHeight();
97  		table.setPreferredScrollableViewportSize(new Dimension(width, height));
98  	}
