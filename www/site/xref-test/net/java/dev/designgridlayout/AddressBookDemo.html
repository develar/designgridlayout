45  	@Override public void go(boolean exitOnClose)
46  	{
47  		_frame = new JFrame(name());
48  		_frame.setName(getClass().getSimpleName());
49  
50  		_frame.setDefaultCloseOperation(exitOnClose
51  			? JFrame.EXIT_ON_CLOSE
52  			: WindowConstants.DISPOSE_ON_CLOSE);
53  
54  		String[] items =
55  		{
56  			"Bunny, Bugs",
57  			"Cat, Sylvester",
58  			"Coyote, Wile E.",
59  			"Devil, Tasmanian",
60  			"Duck, Daffy",
61  			"Fudd, Elmer",
62  			"Le Pew, Pepe",
63  			"Martian, Marvin"
64  		};
65  		JList list = new JList(items);
66  		JScrollPane scroller =
67  			new JScrollPane(
68  				list, ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS,
69  				ScrollPaneConstants.HORIZONTAL_SCROLLBAR_AS_NEEDED);
70  		JPanel listPanel = new JPanel();
71  		listPanel.setLayout(new BorderLayout());
72  		listPanel.add(scroller, BorderLayout.CENTER);
73  		listPanel.setPreferredSize(new Dimension(150, 1));
74  		//TODO: improve margins calculations for List
75  		listPanel.setBorder(BorderFactory.createEmptyBorder(12, 12, 12, 12));
76  
77  		JPanel addressPanel = new JPanel();
78  		DesignGridLayout layout = new DesignGridLayout(addressPanel);
79  		layout.labelAlignment(LabelAlignment.RIGHT);
80  		build(layout);
81  
82  		JPanel top = new JPanel();
83  		top.setName("TOP");
84  		top.setLayout(new BorderLayout());
85  		top.add(listPanel, BorderLayout.LINE_START);
86  		top.add(addressPanel, BorderLayout.CENTER);
87  		_frame.add(top);
88  		prePack();
89  		_frame.pack();
90  		_frame.setLocationRelativeTo(null);
91  		_frame.setVisible(true);
92  	}

111 		JButton saveButton = new JButton("Save");

119 		layout.row().grid(label("State"))		.add(stateField)	.grid(label("Postal Code"))	.add(postalField);
120 		layout.row().grid(label("Country"), 1)	.add(countryField);
121 		layout.emptyRow();
122 		layout.row().center().add(newButton).add(deleteButton).add(editButton).add(saveButton).add(cancelButton);
