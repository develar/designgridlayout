46  	@Override public void go(boolean exitOnClose)
47  	{
48  		_frame = new JFrame(name());
49  		_frame.setName(getClass().getSimpleName());
50  
51  		_frame.setDefaultCloseOperation(exitOnClose
52  		    ? JFrame.EXIT_ON_CLOSE
53  		    : WindowConstants.DISPOSE_ON_CLOSE);
54  
55  		String[] items =
56  		{
57  			"Bunny, Bugs",
58  			"Cat, Sylvester",
59  			"Coyote, Wile E.",
60  			"Devil, Tasmanian",
61  			"Duck, Daffy",
62  			"Fudd, Elmer",
63  			"Le Pew, Pepe",
64  			"Martian, Marvin"
65  		};
66  		JList list = new JList(items);
67  		JScrollPane scroller =
68  		    new JScrollPane(
69  		        list, ScrollPaneConstants.VERTICAL_SCROLLBAR_ALWAYS,
70  		        ScrollPaneConstants.HORIZONTAL_SCROLLBAR_AS_NEEDED);
71  		JPanel listPanel = new JPanel();
72  		listPanel.setLayout(new BorderLayout());
73  		listPanel.add(scroller, BorderLayout.CENTER);
74  		listPanel.setPreferredSize(new Dimension(150, 1));
75  		listPanel.setBorder(BorderFactory.createEmptyBorder(12, 12, 12, 0));
76  
77  		JPanel addressPanel = new JPanel();
78  		DesignGridLayout layout = new DesignGridLayout(addressPanel);
79  		build(layout);
80  
81  		JPanel top = new JPanel();
82  		top.setName("TOP");
83  		top.setLayout(new BorderLayout());
84  		top.add(listPanel, BorderLayout.WEST);
85  		top.add(addressPanel, BorderLayout.CENTER);
86  		_frame.add(top);
87  		prePack();
88  		_frame.pack();
89  		preDisplay();
90  		_frame.setVisible(true);
91  	}

118 		layout.row().grid(label("State"))		.add(stateField)	.grid(label("Postal Code"))	.add(postalField);
119 		layout.row().grid(label("Country"), 1)	.add(countryField);
120 		layout.emptyRow();
121 		layout.row().center().add(newButton).add(deleteButton).add(editButton).add(saveButton).add(cancelButton);
