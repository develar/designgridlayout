202 	private void initListeners()
203 	{
204 		_tree.getSelectionModel().addTreeSelectionListener(new TreeSelectionListener()
205 		{
206 			public void valueChanged(TreeSelectionEvent e)
207 			{
208 				DefaultMutableTreeNode node = (DefaultMutableTreeNode) 
209 					e.getNewLeadSelectionPath().getLastPathComponent();
210 				Object data = node.getUserObject();
211 				if (data instanceof Node)
212 				{
213 					_detail.setCurrentNode((Node) data);
214 				}
215 				else if (data instanceof String)
216 				{
217 					_detail.setCurrentTopic((String) data);
218 				}
219 				else
220 				{
221 					_detail.setCurrentNode(null);
222 				}
223 			}
224 		});
225 		// Listen to double-clicks
226 		_tree.addMouseListener(new MouseAdapter()
227 		{
228 			@Override public void mousePressed(MouseEvent e)
229 			{
230 				if (e.getClickCount() == 2)
231 				{
232 					_detail.launch();
233 				}
234 			}
235 		});
236 	}
