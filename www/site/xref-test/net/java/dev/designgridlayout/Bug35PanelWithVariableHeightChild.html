17  import java.awt.GridBagConstraints;
18  import java.awt.GridBagLayout;
19  import java.awt.Insets;
20  
21  import javax.swing.JPanel;
22  import javax.swing.JScrollPane;
23  import javax.swing.JTextArea;
24  
25  // Issue 35: when DGL form has a JPanel that contains variable-height children
26  public class Bug35PanelWithVariableHeightChild extends AbstractBaseExample
27  {
28  	public static void main(String[] args)
29  	{
30  		Bug35PanelWithVariableHeightChild example = new Bug35PanelWithVariableHeightChild();
31  		example.go(true);
32  	}
33  
34  	@Override public void build(DesignGridLayout layout)
35  	{
36  		JPanel pane = new JPanel(new GridBagLayout());
37  		
38  		GridBagConstraints constraints = new GridBagConstraints();
39  		constraints.anchor = GridBagConstraints.CENTER;
40  		constraints.gridwidth = 1;
41  		constraints.gridheight = 1;
42  		constraints.weightx = 1.0;
43  		constraints.weighty = 1.0;
44  		constraints.insets = new Insets(2, 2, 2, 2);
45  
46  
47  		constraints.gridx = 0;
48  		constraints.gridy = 0;
49  		constraints.fill = GridBagConstraints.BOTH;
50  		pane.add(textarea("a\nb\nc\nd", 3), constraints);
51  
52  		constraints.gridy = 1;
53  		constraints.fill = GridBagConstraints.HORIZONTAL;
54  		pane.add(field("field"), constraints);
55  		
56  		constraints.gridx = 1;
57  		constraints.gridy = 0;
58  		constraints.fill = GridBagConstraints.BOTH;
59  		pane.add(textarea("a\nb\nc\nd", 2), constraints);
60  
61  		constraints.gridy = 1;
62  		pane.add(textarea("a\nb\nc\nd", 4), constraints);
63  		
64  		layout.row().grid().add(pane);
65  	}
66  
67  	protected JScrollPane textarea(String content, int rows)
68  	{
69  		JTextArea area = new JTextArea(rows, 10);
70  		area.setText(content);
71  		return new JScrollPane(area);
72  	}
73  }
